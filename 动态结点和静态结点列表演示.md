```javascript
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>
<body>
    <div id="myDiv" name="div1" class="test1">
        <p id="myPara">test</p>
    </div>
    <script>
        console.log("静态获取结点")
        /* 静态获取结点 （不常使用） */
        /* 获取的是一个结点 */
        const oDiv = document.getElementById('myDiv')
        console.log(oDiv)
        /* 获取的是结点列表 */
        const oDiv1 = document.getElementsByName('div1')
        const oDiv2 = document.getElementsByTagName('div')
        const oDiv3 = document.getElementsByClassName('test1')
        console.log(oDiv1,oDiv2,oDiv3)
        console.log("动态获取结点")
        /* 用 查询选择器  来获取结点（动态获取结点）*/
        /* 获取的是一个结点 */        
        const oDiv4 = document.querySelector('#myDiv')
        console.log(oDiv4)
        // /* 获取的是结点列表 */
        const oDiv5 = document.querySelectorAll('#myDiv')
        const oPara = document.querySelector('#myPara')
        console.log(oDiv5,oPara)
    </script>
</body>
</html>
```
